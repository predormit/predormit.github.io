red-black tree

性质

1.每个节点是红色或是黑色

2.根节点是黑色的

3.所有叶结点是黑色的

4.如果一个结点是红色的，那么它的两个子节点都是黑色的

5.对每个节点，从该节点到所有后代叶结点的简单路径上，均包含相同数量的黑色节点。

作用

使得没有一条路径是另一条的两倍长度，达到二叉平衡树。

旋转

保证二叉树性质，，O(1)

插入

先正常插入，再调整满足性质

1.是树根，变黑

2.p是黑，过

3.p是红的，祖父是黑

（1）叔叔是红

继续去维护上层树的结构

（2）叔叔是黑

将红节点旋转上来，再变黑色


（3）叔叔不存在

同理

删除

先正常删除，如果删去的节点颜色为黑色则调用Fixup

4种情况 1：兄弟结点为红色。做法：兄弟结点改为黑，父亲结点改为红，左旋父亲，更新删除结点的兄弟结点

2:兄弟结点及其孩子都是黑色。做法：兄弟结点改为红色，同时父亲结点为新的删除结点。

3:兄弟结点为黑，左孩子红，右孩子黑。做法：兄弟结点设为红色，兄弟的左孩子设为黑，对兄弟结点右旋，更新删除结点的兄弟结点。

4:兄弟结点为黑，右孩子红。做法：父亲结点的颜色给兄弟结点，父节点再设为黑色，兄弟节点的右孩子为黑，根据父节点左旋，最后设置替换删除结点的结点为根结点，用来终止循环

贴个代码

#include<iostream>
#include<assert.h>

typedef enum{RED = 0,BLACK} Color;
template <typename Type>
struct RBTNode{
    Type key;
    Color color;
    RBTNode* left;
    RBTNode* right;
    RBTNode* parent;
};

template<typename Type>
class RBTree{
public:
    //构造函数
    RBTree(){
        Nil = BuyNode();
        root = Nil;
        Nil -> color = BLACK;
    }
    //析构函数
    ~RBTree(){
        destory(root);
        delete Nil;
        Nil = NULL;
    }
    //中序遍历
    void Inorder(){
        InOrder(root);
        std::cout << std::endl;
    }
    //前序遍历
    void preorder(){
        preOrder(root);
        std::cout << std::endl;
    }
    //插入
    bool Insert(const Type &value){
        RBTNode<Type>* s = root;
        RBTNode<Type>* y = Nil;//记录父节点
        while(s != Nil){
            if(value == s -> key){
                return false;
            }
            y = s;
            if(value > s -> key){
                s = s -> right;
            }
            else{
                s = s -> left;
            }
        }
        s = BuyNode(value);

        if(y == Nil){
            root = s;
            root -> parent = y;
        }
        else{
            if(value < y -> key){
                y -> left = s;
            }
            else{
                y -> right = s;
            }
            s -> parent = y;
        }
        RB_INSERT_FIXUP(s);
        return true;
    }
    //删除
    void Remove(Type key){
        RBTNode<Type>* t;
        if((t = Search(root,key))!= Nil){
            Remove(t);
        }
        else{
            std::cout << "Key is not exist." << std::endl;
        }
    }
    //打印结点颜色
    void InorderPrint(){
        InOrderPrint(root);
    }
protected:
    RBTNode<Type>* BuyNode(const Type &x = Type()){
        RBTNode<Type>* s = new RBTNode<Type>();
        assert(s != NULL);
        s -> color = RED;
        s -> key = x;
        s -> left = s -> right = s -> parent = Nil;
        return s;
    }

    void InOrder(RBTNode<Type>* root){
        if(root != Nil){
            InOrder(root->left);
            std::cout << root -> key << " ";
            InOrder(root->right);
        }
    }

    void preOrder(RBTNode<Type>* root){
        if(root != Nil){
            std::cout << root -> key << " ";
            preOrder(root->left);
            preOrder(root->right);
        }
    }

    void LeftRotate(RBTNode<Type>* x){
        RBTNode<Type>* y = x -> right;
        x -> right = y -> left;
        if(y -> left != Nil){
            y -> left -> parent = x;
        }
        y -> parent = x -> parent;
        if(root == x){
            root = y;
        }
        else if(x == x -> parent -> left){
            x -> parent -> left = y;
        }
        else{
            x -> parent -> right = y;
        }
        y -> left = x;
        x -> parent = y;
    }

    void RightRotate(RBTNode<Type>* x){
        RBTNode<Type>* y = x -> left;
        x -> left = y -> right;
        if(y -> right != Nil){
            y -> right -> parent = x;
        }
        y -> parent = x -> parent;
        if(root == x){
            root = y;
        }
        else if(x == x -> parent -> left){
            x -> parent -> left = y;
        }
        else{
            x -> parent -> right = y;
        }
        y -> right = x;
        x -> parent = y;
    }

    void RB_INSERT_FIXUP(RBTNode<Type>* s){
        while(s->parent->color == RED){
            if(s->parent == s->parent->parent->left){
                RBTNode<Type>* uncle = s->parent->parent->right;
                if(uncle->color==RED){
                    s->parent->color = BLACK;
                    uncle->color = BLACK;
                    s->parent->parent->color = RED;
                    s = s->parent->parent;
                }
                else{
                    if(s == s->parent->right){
                        s = s->parent;
                        LeftRotate(s);//此时s是最左下角的那个红节点
                    }
                    s->parent->color = BLACK;
                    s->parent->parent->color = RED;
                    RightRotate(s->parent->parent);
                }
            }
            else{
                if(s->parent == s->parent->parent->right){
                    RBTNode<Type>* uncle = s->parent->parent->left;
                if(uncle->color==RED){
                    s->parent->color = BLACK;
                    uncle->color = BLACK;
                    s->parent->parent->color = RED;
                    s = s->parent->parent;
                }
                else{
                    if(s == s->parent->left){
                        s = s->parent;
                        RightRotate(s);//此时s是最左下角的那个红节点
                    }
                    s->parent->color = BLACK;
                    s->parent->parent->color = RED;
                    LeftRotate(s->parent->parent);
                }
                }
            }
        }
        root->color = BLACK;
    }
    RBTNode<Type>* Search(RBTNode<Type>* root,Type key)const{
        if(root == Nil){
            return Nil;
        }
        if(root -> key == key){
            return root;
        }
        if(key < root -> key){
            return Search(root->left,key);
        }
        else{
            return Search(root->right,key);
        }
    }
    void Transplant(RBTNode<Type>* u,RBTNode<Type>* v){
        if(u->parent == Nil){
            root = v;
        }
        else if(u == u->parent->left){
            u->parent->left = v;
        }
        else{
            u->parent->right = v;
        }
        v -> parent = u -> parent;
    }
    RBTNode<Type>* Minimum(RBTNode<Type>* x){
        if(x->left == Nil){
            return x;
        }
        return Minimum(x->left);
    }
    void Remove(RBTNode<Type>* z){
        RBTNode<Type>* x = Nil;
        RBTNode<Type>* y = z;
        Color yOriginColor = y->color;
        if(z->left == Nil){
            x = z->right;
            Transplant(z,z->right);
        }
        else if(z->right == Nil){
            x = z->left;
            Transplant(z,z->left);
        }
        else{
            y = Minimum(z->right);
            yOriginColor = y->color;
            x = y->right;
            if(y->parent==z){
                x -> parent = y;
            }
            else{
                Transplant(y,y->right);
                y->right = z->right;
                y->right->parent = y;
            }
            Transplant(z,y);
            y->left = z->left;
            z->left->parent = y;
            y->color = z->color;
        }
        if(yOriginColor == BLACK){
            Remove_Fixup(x);//x是被删结点的子节点
        }
    }
    void Remove_Fixup(RBTNode<Type>* x){
        while(x != root && x->color == BLACK){
            if(x == x->parent->left){
                RBTNode<Type>* w = x->parent->right;
                if(w->color==RED){ //情况1
                    w->color = BLACK;
                    x->parent->color = RED;
                    LeftRotate(x->parent);
                    w = x->parent->right;
                }
                if(w->left->color==BLACK && w->right->color==BLACK){
                    w->color = RED;
                    x = x->parent;
                }//情况2，进下一个while循环
                else{
                    if(w->right->color==BLACK){
                        w->color = RED;
                        w->left->color = BLACK;
                        RightRotate(w);
                        w = x->parent->right;
                    }
                    w->color = x->parent->color;
                    x->parent->color = BLACK;
                    w->right->color = BLACK;
                    LeftRotate(x->parent);
                    x = root; //结束循环
                }
            }
            else{
                RBTNode<Type>* w = x->parent->left;
                if(w->color==RED){ //情况1
                    w->color = BLACK;
                    x->parent->color = RED;
                    RightRotate(x->parent);
                    w = x->parent->left;
                }
                if(w->left->color==BLACK && w->right->color==BLACK){
                    w->color = RED;
                    x = x->parent;
                }//情况2，进下一个while循环
                else{
                    if(w->left->color==BLACK){
                        w->color = RED;
                        w->right->color = BLACK;
                        LeftRotate(w);
                        w = x->parent->left;
                    }
                    w->color = x->parent->color;
                    x->parent->color = BLACK;
                    w->left->color = BLACK;
                    RightRotate(x->parent);
                    x = root; //结束循环
                }
            }
        }
        x->color = BLACK;
    }
    void destory(RBTNode<Type>* &root){
        if(root == Nil){
            return;
        }
        if(root->left != Nil){
            destory(root->left);
        }
        if(root->right != Nil){
            destory(root->right);
        }
        delete root;
        root = NULL;
    }
    void InOrderPrint(RBTNode<Type>* node){
        if(node == Nil){
            return;
        }
        if(node->left != Nil){
            InOrderPrint(node->left);
        }
        std::cout << node->key <<  "(" << ((node->color == BLACK) ? "BLACK" : "RED") << ")" << " ";
        if (node->right != Nil){
            InOrderPrint(node->right);
        }
    }
private:
    RBTNode<Type>* root;
    RBTNode<Type>* Nil;
};

int main(int argc,char* argv[]){
    RBTree<int> rb;
    int arr[] = { 10,7,8,15,5,6,11,13,12};
    int n = sizeof(arr)/sizeof(int);
    for(int i = 0;i < n;i++){
        rb.Insert(arr[i]);
    }
    rb.Inorder();
    std::cout << std::endl;
    rb.preorder();
    std::cout << std::endl;
    rb.InorderPrint();
    std::cout << std::endl;
    rb.Remove(10);
    rb.Inorder();
    rb.Remove(12);
    std::cout << std::endl;
    rb.InorderPrint();
}
